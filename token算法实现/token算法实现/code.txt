int main() {
    std::ifstream file("example_code.txt"); // 打开文件
    if (!file.is_open()) {
        std::cerr << "无法打开文件。" << std::endl;
        return 1;
    }

    std::string code;
    std::string line;
    while (std::getline(file, line)) { // 逐行读取文件
        code += line + "\n"; // 将每行追加到代码字符串中
    }

    file.close(); // 关闭文件

    // 对代码进行标记
    std::vector<TOKEN> tokens = scan(code);

    // 输出标记
    for (const auto& token : tokens) {
        std::cout << "mark: " << token.word << ", type: " << to_chinese(token.code) << std::endl;
    }

    return 0;
}
