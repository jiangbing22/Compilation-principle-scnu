
        std::cerr << "can not open" << std::endl;
        return 1;

    std::string code;
    std::string line;
    while (std::getline(file, line)) { // 逐行读取文件
        code += line + "\n"; // 将每行追加到代码字符串中
    }

    file.close();
    std::cout << code;

    // 对代码进行标记
    std::vector<token::TOKEN> tokens = token::scan(code);

    // 输出标记
    for (const auto& token : tokens) {
        std::cout << "标记: " << token.word << ", 类型: " << to_chinese(token.code) << std::endl;
    }

    return 0;
}
